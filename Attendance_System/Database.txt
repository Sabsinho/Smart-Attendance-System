CREATE DATABASE smart_attendance;

CREATE TABLE alerts (
    id INT(11) NOT NULL AUTO_INCREMENT,
    student_id CHAR(5) NOT NULL,
    course_id INT(11) NOT NULL,
    alert_message TEXT NOT NULL,
    resolved TINYINT(1) DEFAULT 0,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id),
    INDEX (course_id)
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COLLATE=latin1_swedish_ci;


CREATE TABLE attendance (
    id INT(11) NOT NULL AUTO_INCREMENT,
    student_id CHAR(5) NOT NULL,
    course_id INT(11) NOT NULL,
    attendance_date DATE DEFAULT NULL,
    attendance_time TIME DEFAULT NULL,
    status ENUM('Present', 'Absent', 'Late') DEFAULT 'Present',
    date DATE DEFAULT NULL,
    time TIME DEFAULT NULL,
    PRIMARY KEY (id),
    INDEX (student_id),
    INDEX (course_id),
    INDEX (attendance_date)
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COLLATE=latin1_swedish_ci;



CREATE TABLE courses (
    id INT(11) NOT NULL AUTO_INCREMENT,
    course_name VARCHAR(100) NOT NULL,
    course_code VARCHAR(20) NOT NULL,
    semester_id INT(11) NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id),
    INDEX (semester_id)
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COLLATE=latin1_swedish_ci;



CREATE TABLE course_faculty (
    id INT(11) NOT NULL AUTO_INCREMENT,
    course_id INT(11) NOT NULL,
    faculty_id INT(11) NOT NULL,
    PRIMARY KEY (id),
    INDEX (course_id),
    INDEX (faculty_id)
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COLLATE=latin1_swedish_ci;


CREATE TABLE faculties (
    id INT(11) NOT NULL AUTO_INCREMENT,
    faculty_name VARCHAR(100) NOT NULL,
    PRIMARY KEY (id),
    INDEX (faculty_name)
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COLLATE=latin1_swedish_ci;



CREATE TABLE lecturers (
    id INT(11) NOT NULL AUTO_INCREMENT,
    first_name VARCHAR(100) DEFAULT NULL,
    last_name VARCHAR(100) DEFAULT NULL,
    email VARCHAR(100) DEFAULT NULL,
    phone VARCHAR(20) DEFAULT NULL,
    staff_id VARCHAR(20) DEFAULT NULL,
    course_id INT(11) DEFAULT NULL,
    portrait_path VARCHAR(255) DEFAULT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    faculty_id INT(11) NOT NULL,
    shift_id INT(11) NOT NULL,
    PRIMARY KEY (id),
    INDEX (staff_id),
    INDEX (course_id)
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COLLATE=latin1_swedish_ci;




CREATE TABLE semesters (
    id INT(11) NOT NULL AUTO_INCREMENT,
    semester_name VARCHAR(50) NOT NULL,
    start_date DATE NOT NULL,
    end_date DATE NOT NULL,
    PRIMARY KEY (id)
);

CREATE TABLE shifts (
    id INT(11) NOT NULL AUTO_INCREMENT,
    shift_name ENUM('Morning', 'Afternoon', 'Evening') NOT NULL,
    PRIMARY KEY (id),
    INDEX (shift_name)
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COLLATE=latin1_swedish_ci;


CREATE TABLE students (
    id INT(11) NOT NULL AUTO_INCREMENT,
    student_id CHAR(5) NOT NULL,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    enrollment_year YEAR(4) NOT NULL,
    semester_id INT(11) NOT NULL,
    faculty_id INT(11) NOT NULL,
    shift_id INT(11) NOT NULL,
    phone VARCHAR(20) DEFAULT NULL,
    address TEXT DEFAULT NULL,
    portrait_path VARCHAR(255) DEFAULT NULL,
    encoding_path VARCHAR(255) DEFAULT NULL,
    status ENUM('Active', 'Suspended', 'Graduated', 'Withdrawn') DEFAULT 'Active',
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id),
    INDEX (student_id),
    INDEX (semester_id),
    INDEX (faculty_id),
    INDEX (shift_id)
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COLLATE=latin1_swedish_ci;



CREATE TABLE system_logs (
    id INT(11) NOT NULL AUTO_INCREMENT,
    user_id INT(11) NOT NULL,
    action TEXT NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id),
    INDEX (user_id)
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COLLATE=latin1_swedish_ci;


CREATE TABLE users (
    id INT(11) NOT NULL AUTO_INCREMENT,
    username VARCHAR(50) NOT NULL,
    password VARCHAR(255) NOT NULL,
    role ENUM('admin', 'lecturer') NOT NULL,
    lecturer_id INT(11) DEFAULT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id),
    INDEX (username)
) ENGINE=InnoDB DEFAULT CHARSET=latin1 COLLATE=latin1_swedish_ci;
